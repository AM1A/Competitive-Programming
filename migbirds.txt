int migratoryBirds(int n/*, int ar_size*/, int* ar)
{
    // Complete this function
    int arr[5] = {0,0,0,0,0};
    int i,max = -1,maxi,ret;
    for(i = 0;i<n;i++)
    {
        arr[ar[i]%5]++;
    }
    for(i = 4;i>=0;i--)
    {
        if(arr[i]>=max)
        {
            max = arr[i];
            maxi = i;
        }
    }
    if(maxi!=0)
    {
        ret = maxi;
    }
    else
    {
        ret = 5;
    }
    return ret;
    
}

int main() {
    int n;

    for(int ar_i = 0; ar_i < n; ar_i++){
       scanf("%i",&ar[ar_i]);
    }
    int result = migratoryBirds(n, ar);

}
