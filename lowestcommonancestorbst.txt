/*
Node is defined as 

typedef struct node
{
   int data;
   node * left;
   node * right;
}node;

*/


node * lca(node * root, int v1,int v2)
{
    if(v1>root->data&v2>root->data)
    {
        root = lca(root->right,v1,v2);
    }
    else if(v1<root->data&v2<root->data)
    {
        root = lca(root->left,v1,v2);
    }
    else
    {   
        root = root;
    }
    return root;

   
}
